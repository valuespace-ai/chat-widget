ValueSpace Chat Widget - External Usage Example

This page demonstrates how to integrate the ValueSpace Chat Widget into any website using an iframe for better isolation.

Basic Integration

Include the script directly in your HTML and initialize the widget:

<script src="path/to/vs-chat-widget.iife.js"></script>
<script>
  // Initialize with default settings
  const chatWidget = window.initChatWidget({
    botServiceUrl: 'http://localhost:3978',
    showTestButtons: true // Enable test buttons regardless of environment
  });
</script>

iframe Integration for Better Isolation

For better isolation and to prevent CSS conflicts, you can load the widget in an iframe:

<!-- Create iframe for chat widget -->
<iframe id="chat-widget-iframe" style="border:none; position:fixed; bottom:0; right:0; width:100%; height:100%; z-index:999; pointer-events:none;"></iframe>

<script>
  function initChatWidgetInIframe(config) {
    const iframe = document.getElementById('chat-widget-iframe');
    
    // Set up the iframe content
    iframe.onload = function() {
      // Add widget script to iframe
      const script = iframe.contentDocument.createElement('script');
      script.src = "path/to/vs-chat-widget.iife.js";
      script.onload = function() {
        // Add development flag
        iframe.contentWindow.process = { env: { NODE_ENV: "development" } };
        // Initialize widget with config
        iframe.contentWindow.initChatWidget(config);
      };
      iframe.contentDocument.body.appendChild(script);
      
      // Add CSS to make widget elements clickable
      const style = iframe.contentDocument.createElement('style');
      style.textContent = `
        body { margin: 0; padding: 0; }
        .webchat__bubble, .vs-chat-invitation, .vs-chat-launcher, 
        .vs-chat-notification, .webchat__send-box {
          pointer-events: auto !important;
        }
      `;
      iframe.contentDocument.head.appendChild(style);
    };
    
    // Create empty HTML document in iframe
    iframe.srcdoc = '<!DOCTYPE html><html><head><meta charset="UTF-8"></head><body><div id="vs-chat-widget-container"></div></body></html>';
    
    return {
      destroy: function() { iframe.srcdoc = ''; },
      sendProactiveMessage: function(text, options) {
        if (iframe.contentWindow.chatWidgetProactive) {
          iframe.contentWindow.chatWidgetProactive.sendProactiveMessage(text, options);
        }
      },
      showNotification: function(text, options) {
        if (iframe.contentWindow.chatWidgetProactive) {
          iframe.contentWindow.chatWidgetProactive.showNotification(text, options);
        }
      },
      openChat: function() {
        if (iframe.contentWindow.chatWidgetProactive) {
          iframe.contentWindow.chatWidgetProactive.openChat();
        }
      },
      showWelcomePopup: function() {
        if (iframe.contentWindow.chatWidgetProactive) {
          iframe.contentWindow.chatWidgetProactive.showWelcomePopup();
        }
      }
    };
  }
</script>
